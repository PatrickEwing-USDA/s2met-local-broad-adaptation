.specialist_test = function(x) {
  nc = ncol(x)
  obs = x[,1]
  pr = sweep(x, 1, obs, '>=')
  out = rowSums(pr)/nc
  out
}
.generalist_test = function(x) {
  nc = ncol(x)
  obs = x[,1]
  pr = sweep(x, 1, obs, '<')
  out = rowSums(pr)/nc
  out
}
.matrix_quantile = function(x, q) {
  apply(x, 1, quantile, q)
}
